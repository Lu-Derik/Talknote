src/extension.ts (示例实现，替换或合并到你的扩展入口文件):
```typescript
import * as vscode from 'vscode';

export function activate(context: vscode.ExtensionContext) {
  const recordCmd = vscode.commands.registerCommand('talknote.record', async () => {
    const config = vscode.workspace.getConfiguration('talknote');
    const separator = config.get<string>('separator', '------------------------------');
    const requireAt = config.get<boolean>('requireAtMention', true);

    const input = await vscode.window.showInputBox({ prompt: 'Enter talk content (include @talknot if required)' });
    if (!input) {
      return;
    }
    if (requireAt && !input.includes('@talknot')) {
      vscode.window.showWarningMessage('Talk not recorded: missing @talknot mention.');
      return;
    }

    const doc = vscode.window.activeTextEditor?.document;
    const edit = new vscode.WorkspaceEdit();
    const textToInsert = `${separator}\n${input}\n${separator}\n`;
    if (doc && vscode.window.activeTextEditor) {
      const pos = vscode.window.activeTextEditor.selection.active;
      edit.insert(doc.uri, pos, textToInsert);
      await vscode.workspace.applyEdit(edit);
      vscode.window.showInformationMessage('Talk recorded.');
    } else {
      // fallback: copy to clipboard
      await vscode.env.clipboard.writeText(textToInsert);
      vscode.window.showInformationMessage('No active editor — talk copied to clipboard.');
    }
  });

  context.subscriptions.push(recordCmd);
}

export function deactivate() {}
```

如何测试：
- 在扩展工程按 F5 启动 Extension Development Host。
- 在 Dev Host 中打开命令面板运行 Talknote: Record Talk（或快捷键绑定）。
- 可在 Settings（UI 或 settings.json）修改 talknote.separator 即时生效。
